Set 10

 

1.    create table student(sid varchar(10) primary key, sname char(30), address varchar(30), mobno varchar(30), agg number, check(agg >= 10));

 

create table book(bid varchar(10) primary  key, bname char(30), sid varchar(10), isdate date, vdate date, foreign key(sid) references student(sid));

 

2.    create view viewtitle as select S.sname as Student, S.address, S.mobno as contact from student S, book B where B.sid = S.sid and B.bname = 'Database concepts';

 

3.    select * from book order by isdate;

 

4.    select sname from student s where sid in (select sid from book group by sid having count(sid) = 2)

 

here we used having clause to use aggregate conditions and also remember to use group by clause

 

5.    declare

bd varchar(30);

1.    create table doctor(did varchar(10) primary key, dname char(20), spec char(20), xp number, contact char(30));

 

create table patient(pid varchar(10) primary key, pname char(30), typ char(10), dtag varchar(10), age number, adm date, foreign key(dtag) references doctor(did));

 

alter table patient add constraints f1 check(typ = 'IN' or typ = 'OUT');

 

2.    create view v1 as (select P.pid, P.pname from patient P, Doctor D where P.dtag = D.did and (to_date('12-31-2022', 'MM-DD-YYYY') - to_date(P.ADM, 'MM-DD-YYYY') > 30) and D.SPEC = 'ORTHO')

select * from v1

 

3.    select pid, pname, adm from patient order by adm

 

4.    select D.did, D.dname from doctor D where D.did in (select dtag from patient group by dtag having count(pid) = 2)
